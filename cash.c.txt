#include <stdio.h>
int Calculate_quarters(int change);
int Calculate_dimes(int change);
int Calculate_nickels(int change);
int Calculate_pennies(int change);

int main(void)
{
    int change;

    do{
       change = get_int("Whats the change?: ");

    } while (change < 1);

    int quarters = Calculate_quarters(change);
    change = change - (quarters * 25);
    //printf("%i quarters\n", quarters);

    int dimes = Calculate_dimes(change);
    change = change - (dimes * 10);
    //printf("%i dimes\n", dimes);

    int nickels = Calculate_nickels(change);
    change = change - (nickels * 5);
    //printf("%i nickels\n", nickels);

    int pennies = Calculate_pennies(change);
    change = change - (pennies * 1);
    //printf("%i pennies\n", pennies);



    int total_change = (quarters + dimes + nickels + pennies);


    printf("%i coins total\n", total_change);
}


int Calculate_quarters(int change)
{
    int quarters = 0;
    while (change >= 25)
    {
        quarters++;
        change = change - 25;
    }
    return quarters;
}

int Calculate_dimes(int change)
{
    int dimes = 0;
    while (change >= 10)
    {
        dimes++;
        change = change - 10;
    }
    return dimes;
}

int Calculate_nickels(int change)
{
    int nickels = 0;
    while(change >= 5)
    {
        nickels++;
        change = change - 5;
    }
    return nickels;
}
int Calculate_pennies(int change)
{
    int pennies = 0;
    while(change >=1)
    {
        pennies++;
        change = change - 1;
    }
    return pennies;
}
